[docs]def solve_spd_chol(L, b):
    """
    Solves a symmetric positive definite system A x = b
    where A = L L'
    """
    # We have to solve L L' x = b
    # We first solve L u = b
    u = solve_Lx_b(L, b)
    # We now solve L' x = u
    x = solve_LTx_b(L, u)
    return x
